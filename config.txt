# Delimiter:
#  tab (fields separated by tab, repetitions by newline)
#  newline (fields and repetitions separated by newline)
#
# Block order:
#   from_top
#   from_bottom
#   oldest_first
#   newest_first
#   custom (specify one by one at runtime)
#
# Repetition order:
#   oldest_first
#   newest_first
#
# Ignore sentences: true/false
#
# Field order:
#   0 - blank field
#   1 - polish translation
#   2 - english phrase
#   3 - synonyms
#   4 - note
#
# Max workers:
#   number of threads used for downloading blocks of repetitions
#   (more threads = faster download)
#   (do not use more than 6)

delimiter=tab
block_order=newest_first
repetition_order=newest_first
ignore_sentences=false
field_order=1,2,3,4
max_workers=4
